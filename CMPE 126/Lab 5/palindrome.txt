Function check_palindrome(n, start = 0, end = -1):
	If n < 0:
		Return false

	If end == -1:
		If n == 0:
			end = 0
		Else:
			tmp = n
			digits = 0
			While tmp > 0:
				tmp = tmp / 10
				digits = digits + 1
			end = digits - 1

	If start >= end:
		Return true

	powEnd = 10^end
	powStart = 10^start
	firstDigit = (n / powEnd) % 10
	lastDigit  = (n / powStart) % 10

	If firstDigit != lastDigit:
		Return false

	Return check_palindrome(n, start + 1, end - 1)

End Function
